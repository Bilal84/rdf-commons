18  
19  import org.sindice.rdfcommons.model.TripleIterator;
20  import org.sindice.rdfcommons.model.TripleSet;
21  
22  
23  /**
24   * Defines the triples storage.
25   *
26   * @see org.sindice.rdfcommons.model.TripleFilter
27   * @author Michele Mostarda ( mostarda@fbk.eu )
28   * @version $Id: TripleStorage.java 24 2010-12-01 23:47:33Z michele.mostarda $
29   */
30  public interface TripleStorage {
31  
32      /**
33       * Opens the triple storage.
34       *
35       * @param config configuration for the triple storage.
36       */
37      void openStorage(TripleStorageConfig config);
38  
39      /**
40       * Closes the triple storage.
41       */
42      void closeStorage();
43  
44      /**
45       * Checks if a triple storage is open or not.
46       *
47       * @return <code>true</code> if opened, <code>false</code> otherwise.
48       */
49      boolean isOpen();
50  
51      /**
52       * Stores a triple set in the default graph.
53       *
54       * @param ts
55       * @throws StorageException if an error occurs during storaging.
56       */
57      void store(TripleSet ts) throws StorageException;
58  
59      /**
60       * Stores the triple set in the named <i>graph</i>.
61       *
62       * @param graph name of the graph.
63       * @param ts
64       * @throws StorageException
65       */
66      void store(String graph, TripleSet ts) throws StorageException;
67  
68      /**
69       * Removes the triples listed in the triple set from the default graph.
70       * 
71       * @param ts list of triples to be removed.
72       * @throws StorageException if an error occurs during the triples removal.
73       */
74      void remove(TripleSet ts) throws StorageException;
75  
76      /**
77       * Removes the triples listed in the triple set from the named <i>graph</i>.
78       *
79       * @param graph name of the graph.
80       * @param ts
81       * @throws StorageException
82       */
83      void remove(String graph, TripleSet ts) throws StorageException;
84  
85      /**
86       * Returns an iterator over the triples matching the filter searched in the default graph.
87       *
88       * @param tripleFilter the filter to be applied on the triples. 
89       * @return all the triples matching the filter.
90       */
91      TripleIterator getTriplesIterator(TripleStorageFilter tripleFilter);
92  
93      /**
94       * Returns an iterator over the triples matching the filter searched in the named <i>graph</i>.
95       *
96       * @param graph name of the graph.
97       * @param tripleFilter
98       * @return all the triples matching the filter.
99       */
100     TripleIterator getTriplesIterator(String graph, TripleStorageFilter tripleFilter);
101 
102     /**
103      * Returns a set of triples matching the triple filter searched in the default graph.
104      *  
105      * @param tripleFilter
106      * @return set of triples matching with filter criteria.
107      */
108     TripleSet getTriples(TripleStorageFilter tripleFilter);
109 
110     /**
111      * Returns a set of triples matching the triple filter searched in the named <i>graph</i>.
112      *
113      * @param graph
114      * @param tripleFilter
115      * @return set of triples matching with filter criteria.
116      */
117     TripleSet getTriples(String graph, TripleStorageFilter tripleFilter);
118 
119     /**
120      * Returns the <i>SPARQL</i> endpoint of the triple storage.
121      *
122      * @return  the endpoint.
123      */
124     SparqlEndPoint getSparqlEndPoint();
125 
126     /**
127      * Removes the entire content of the named <i>graph</i>.
128      */
129     void clear(String graph);
130 
131     /**
132      * Removes the content of the entire storage.
133      */
134     void clearAll();
135 
136 }
