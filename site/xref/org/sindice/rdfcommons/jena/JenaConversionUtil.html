49      /**
50       * Converts a <i>Jena</i> triple in a {@link org.sindice.rdfcommons.model.Triple}.
51       *
52       * @param triple Jena triple to convert.
53       * @return the equivalent {@link org.sindice.rdfcommons.model.Triple}.
54       */
55      @SuppressWarnings("unchecked")
56      public static Triple convertJenaTripleToTriple(com.hp.hpl.jena.graph.Triple triple) {
57          final Node subject = triple.getSubject();
58          final Node object  = triple.getObject();
59  
60          final String subjectStr;
61          if(subject.isURI()) {
62              subjectStr = subject.getURI();
63          } else {
64              subjectStr = subject.getBlankNodeLabel();
65          }
66  
67          final Object objectValue;
68          if(object.isURI()) {
69              objectValue = object.getURI();
70          } else if(object.isBlank()) {
71              objectValue = object.getBlankNodeLabel();
72          } else {
73              objectValue = object.getLiteralValue();
74          }
75  
76          return new Triple(
77                  subjectStr,
78                  triple.getPredicate().getURI(),
79                  objectValue,
80                  subject.isBlank() ? SubjectType.bnode : SubjectType.uri,
81                  getObjectType(object)
82          );
83      }

85      /**
86       * Converts a {@link org.sindice.rdfcommons.model.TripleSet} to a list of <i>Jena</i> triples.
87       *
88       * @param tripleSet input triple set.
89       * @return list of converted Jena triples.
90       */
91      public static com.hp.hpl.jena.graph.Triple[] convertToJenaTriples(TripleSet tripleSet) {
92          com.hp.hpl.jena.graph.Triple[] result = new com.hp.hpl.jena.graph.Triple[ tripleSet.getSize() ];
93          int i = 0;
94          for(Triple triple : tripleSet) {
95              result[i] = new com.hp.hpl.jena.graph.Triple(
96                  createSubjectNode(triple),
97                  Node.createURI( triple.getPredicate() ),
98                  createObjectNode(triple)
99              );
100             i++;
101         }
102         return result;
103     }

105     /**
106      * Converts a <i>Jena</i> model in a {@link org.sindice.rdfcommons.model.TripleSet}.
107      *
108      * @param model input model.
109      * @return generated triple set.
110      */
111     public static TripleSet convertJenaModelToTripleSet(com.hp.hpl.jena.rdf.model.Model model) {
112         StmtIterator iter = model.listStatements();
113         TripleSet result = new TripleBuffer();
114         while(iter.hasNext()) {
115             Statement statement = iter.nextStatement();
116             result.addTriple( convertJenaTripleToTriple( statement.asTriple() ) );
117         }
118         return result;
119     }

121     /**
122      * Converts a {@link org.sindice.rdfcommons.model.TripleSet} to a Jena {@link com.hp.hpl.jena.rdf.model.Model}.
123      *
124      * @param ts triple set.
125      * @return generated model.
126      */
127     public static com.hp.hpl.jena.rdf.model.Model convertTripleSetToJenaModel(TripleSet ts) {
128         Model model = ModelFactory.createDefaultModel();
129         for(Triple t : ts) {
130             com.hp.hpl.jena.rdf.model.Resource subject;
131             if( t.isSubjectBNode() ) {
132                 subject = model.createResource(AnonId.create( t.getSubject() ) );
133             } else {
134                 subject = model.createResource( t.getSubject() );
135             }
136 
137             com.hp.hpl.jena.rdf.model.Property predicate = model.createProperty( t.getPredicate() );
138 
139             com.hp.hpl.jena.rdf.model.RDFNode object;
140             if( t.isObjectBNode() ) {
141                 object = model.createResource(AnonId.create( t.getObjectAsString() ) );
142             } else {
143                 if(t.isObjLiteral()) {
144                     object = model.createTypedLiteral( t.getObject() );
145                 } else {
146                     object = model.createResource( t.getObjectAsString() );
147                 }
148             }
149 
150             model.add(subject, predicate, object);
151         }
152         return model;
153     }

155     /**
156      * Creates an appropriate <i>Jena</i> node representing the subject of the given triple.
157      *
158      * @param triple input triple.
159      * @return subject nodel.
160      */
161     public static com.hp.hpl.jena.graph.Node createSubjectNode(Triple triple) {
162         return
163                 triple.isSubjectBNode()
164                         ?
165                 Node.createAnon( AnonId.create(triple.getSubject()) ) : Node.createURI( triple.getSubject() );
166     }

168     /**
169      * Creates an appropriate <i>Jena</i> node representing the object of the given triple.
170      *
171      * @param triple input triple.
172      * @return object nodel.
173      */
174 
175     public static com.hp.hpl.jena.graph.Node createObjectNode(Triple triple) {
176         if(triple.isObjLiteral()) {
177             return Node.createLiteral( triple.getObject().toString(), null, getRDFDatatype(triple.getObject()) );
178         }
179         if(triple.isObjectBNode()) {
180             return Node.createAnon(AnonId.create( triple.getObjectAsString() ) );
181         }
182         return Node.createURI( triple.getObjectAsString() );
183     }

185     /**
186      * converts a {@link org.sindice.rdfcommons.storage.TripleStorageFilter} in a Jena <i>TripleMatch</i>.
187      *
188      * @param filter input filter.
189      * @return generated triple match.
190      */
191     public static com.hp.hpl.jena.graph.TripleMatch convertToJenaTripleMatch(final TripleStorageFilter filter) {
192         return new TripleMatch() {
193 
194             public Node getMatchSubject() {
195                 final String subjectMatch = filter.getSubjectMatching();
196                 if( filter.requireSubjectBlank() ) {
197                     return
198                             subjectMatch == null
199                                     ?
200                             Node.createAnon()
201                                     :
202                             Node.createAnon( AnonId.create(subjectMatch) );
203                 }
204                 return subjectMatch == null ? null : Node.createURI(subjectMatch);
205             }
206 
207             public Node getMatchPredicate() {
208                 final String predicateMatch = filter.getPredicateMatching();
209                 return predicateMatch == null ? null : Node.createURI(predicateMatch);
210             }
211 
212             public Node getMatchObject() {
213                 final Object objectMatch = filter.getObjectMatching();
214                 if( filter.requireObjectBlank() ) {
215                     return
216                             objectMatch == null
217                                 ?
218                             Node.createAnon()
219                                 :
220                             Node.createAnon( AnonId.create(objectMatch.toString()) );
221                 }
222                 return filter.requireLiteral()
223                     ?
224                     objectMatch == null ? null : Node.createLiteral(
225                             objectMatch.toString(),
226                             null,
227                             getRDFDatatype(objectMatch)
228                     )
229                     :
230                     objectMatch == null ? null : Node.createURI( objectMatch.toString() );
231             }
232 
233             public com.hp.hpl.jena.graph.Triple asTriple() {
234                 return new com.hp.hpl.jena.graph.Triple( getMatchSubject(), getMatchPredicate(), getMatchObject() );
235             }
236         };
237     }

250     /**
251      * Converts a {@link org.sindice.rdfcommons.storage.TripleStorageFilter} in a <i>SPARQL Construct</i> query.
252      *
253      * @param tsf
254      * @return the SPARQL query.
255      */
256     public static String convertToSparqlConstructQuery(String graphName, TripleStorageFilter tsf) {
257         return String.format(
258                 "PREFIX xsd: <http://www.w3.org/2001/XMLSchema#> " +
259                 "CONSTRUCT {%s %s %s} " +
260                 "FROM <%s> " +
261                 "WHERE {%s %s %s %s %s %s}",
262 
263                 toTerm( "?s", tsf.getSubjectMatching()   ),
264                 toTerm( "?p", tsf.getPredicateMatching() ),
265                 toObjectTerm( "?o", tsf.getObjectMatching() ),
266 
267                 graphName,
268 
269                 toTerm( "?s", tsf.getSubjectMatching()   ),
270                 toTerm( "?p", tsf.getPredicateMatching() ),
271                 toObjectTerm( "?o", tsf.getObjectMatching() ),
272 
273                 (tsf.getSubjectMatching() == null && tsf.requireSubjectBlank()) ? ". FILTER isBlank(?s)"   : "",
274                 (tsf.getObjectMatching()  == null && tsf.requireObjectBlank())  ? ". FILTER isBlank(?o)"   : "",
275                 (tsf.getObjectMatching()  == null && tsf.requireLiteral())      ? ". FILTER isLiteral(?o)" : ""
276         );
277     }

279     /**
280      * Returns the most appropriate <i>RDF Datatype</i> for the given object.
281      *
282      * @param o input object.
283      * @return the datatype.
284      * @throws IllegalArgumentException if the object is not supported.
285      */
286     public static RDFDatatype getRDFDatatype(Object o) {
287         if(o instanceof String) {
288             return XSDDatatype.XSDstring;
289         }
290         if(o instanceof Integer) {
291             return XSDDatatype.XSDinteger;
292         }
293         if(o instanceof Long) {
294             return XSDDatatype.XSDlong;
295         }
296         if(o instanceof Short) {
297             return XSDDatatype.XSDshort;
298         }
299         if(o instanceof Float) {
300             return XSDDatatype.XSDfloat;
301         }
302         if(o instanceof Double) {
303             return XSDDatatype.XSDdouble;
304         }
305         if(o instanceof Byte) {
306             return XSDDatatype.XSDbyte;
307         }
308         if(o instanceof Boolean) {
309             return XSDDatatype.XSDboolean;
310         }
311         if(o instanceof Character) {
312             return XSDDatatype.XSDstring;
313         }
314         if(o instanceof Date) {
315             return XSDDatatype.XSDdateTime;
316         }
317         throw new IllegalArgumentException("Invalid object type: " + o);
318     }

320     private static String toTerm(String var, String in) {
321         return in == null ? var : String.format("<%s>", in);
322     }

324     private static String getRDFDatatypeFragment(Object in) {
325         String uri = getRDFDatatype(in).getURI();
326         final int indexofFragment = uri.indexOf('#');
327         if(indexofFragment == -1) {
328             throw new IllegalArgumentException();
329         }
330         return uri.substring( indexofFragment + 1 );
331     }
332 
333     private static String toObjectTerm(String var, Object in) {
334         return in == null ? var : String.format("\"%s\"^^xsd:%s", in.toString(), getRDFDatatypeFragment(in) );
335     }
336 
337     private static Triple.ObjectType getObjectType(Node node) {
338         if(node.isLiteral()) {
339             return ObjectType.literal;
340         }
341         if(node.isBlank()) {
342             return ObjectType.bnode;
343         }
344         return ObjectType.uri;
345     }
