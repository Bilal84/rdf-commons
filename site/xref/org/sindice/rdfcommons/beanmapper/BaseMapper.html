106      *
107      * @param in the class of the type.
108      * @return <code>true</code> if the type is primitive, <code>false</code> otherwise.

110     protected static boolean isPrimitive(Class in) {
111         if(definesInterface(in, Literal.class)) {
112             return true;
113         }
114         for(Class clazz : PRIMITIVE_TIPES) {

117             }
118         }
119         return false;

142     /**
143      * Converts a literal string to the given type.
144      *
145      * @param literal the literal to be converted.
146      * @param type the destination type.
147      * @return the converted value.
148      * @throws IllegalArgumentException if a conversion error occurs.
149      * @throws IllegalArgumentException if the given <i>type</i> is not supported.
150      */
151     protected static Object literalAsPrimitive(String literal, Class type) {
152         if(String.class.equals(type)) {
153             return literal;
154         }
155         if(Boolean.class.equals(type) || boolean.class.equals(type)) {
156             return Boolean.parseBoolean(literal);
157         }
158         if(Short.class.equals(type) || short.class.equals(type)) {
159             return Short.parseShort(literal);
160         }
161         if(Integer.class.equals(type) || int.class.equals(type)) {
162             return Integer.parseInt(literal);
163         }
164         if(Long.class.equals(type) || long.class.equals(type)) {
165             return Long.parseLong(literal);
166         }
167         if(Float.class.equals(type) || float.class.equals(type)) {
168             return Float.parseFloat(literal);
169         }
170         if(Double.class.equals(type) || double.class.equals(type)) {
171             return Double.parseDouble(literal);
172         }
173         if(Date.class.equals(type)) {
174             try {
175                 return simpleDateFormat.parse(literal);
176             } catch (ParseException e) {
177                 throw new IllegalArgumentException( String.format("Expected a Date for literal '%s'.", literal) );
178             }
179         }
180         if(URL.class.equals(type)) {
181             try {
182                 return new URL(literal);
183             } catch (MalformedURLException murle) {
184                 throw new IllegalArgumentException( String.format("Expected a URL for literal '%s'.", literal) );
185             }
186         }
187         throw new UnsupportedOperationException(
188                 String.format("Cannot convert literal %s to type %s", literal, type)
189         );
190     }

192     /**
193      * Returns the URL representing the class.
194      *
195      * @param clazz
196      * @return URL string representing the class and package.
197      */
198     protected static String getClassURL(Class clazz) {
199         String subject = findSubjectAnnotation(clazz);
200         if(subject != null) {
201             return subject;
202         }
203         String packageName = clazz.getPackage() != null ? clazz.getPackage().getName() : "";
204         return String.format("http://%s/%s", packageName, clazz.getSimpleName());
205     }

207     /**
208      * Generates the URL representing an instance URL.
209      *
210      * @param o object to be URLified.
211      * @return string containing URL.
212      */
213     public static String getObjectURL(Object o) {
214         Method target = null;
215         for(Method method : o.getClass().getMethods()) {
216             if(method.getAnnotation(Id.class) != null) {
217                 target = method;
218                 break;
219             }
220         }
221 
222         if(target == null) {
223             return String.format("%s#%s", SerializerVocabulary.INSTANCE, Integer.toHexString( o.hashCode()) );
224         }
225 
226         if(target.getParameterTypes().length != 0) {
227             throw new IllegalArgumentException(
228                     "The method " + target + " to be annotated with " + Id.class + " must accept 0 parameters."
229             );
230         }
231         try {
232             return target.invoke(o).toString();
233         } catch (Exception e) {
234             throw new RuntimeException("Error while retrieving object identifier from method " + target, e);
235         }
236     }

238     /**
239      * Returns the URL representing a property.
240      *
241      * @param classUrl
242      * @param property
243      * @param method the method associated to the property.
244      * @return URL string representing property.
245      */
246     protected static String getPropertyURL(String classUrl, String property, Method method) {
247         String propertyAnnotation = findPropertyAnnotation(method);
248         if(propertyAnnotation != null) {
249             return propertyAnnotation;
250         }
251         String namespaceAnnotation = findNamespaceAnnotation( method.getDeclaringClass() );
252         if(namespaceAnnotation != null) {
253             return String.format("%s/%s", namespaceAnnotation, property);
254         }
255         return String.format("%s#%s", classUrl, property);
256     }

266         if(key.isLiteral()) {
267             return buffer.addBNodeSubjectTriple(SerializerVocabulary.KEY, key.getId(), Triple.ObjectType.literal);

269             return (String) key.getId();

348     /**
349      * Finds the property string possibly defined for a method.
350      *
351      * @param method
352      * @return the property string.
353      * @throws java.net.MalformedURLException if the property string is not an URL.
354      */
355     protected static String findPropertyAnnotation(Method method) {
356         Property property = (Property) findAnnotation(method, Property.class);
357         if(property == null) {
358             return null;
359         }
360         try {
361             return new URL( property.value() ).toString();
362         } catch (MalformedURLException murle) {
363             throw new RuntimeException(
364                     String.format(
365                         "Invalid annotation for method %s, the %s annotation value must be a valid URL.",
366                         method,
367                         Property.class
368                     )
369             );
370         }
371     }
