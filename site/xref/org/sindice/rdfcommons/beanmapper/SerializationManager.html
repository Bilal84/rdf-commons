28  /**
29   * This class performs the <i>bean<i> to <i>RDF</i> conversion.
30   *
31   * @author Michele Mostarda ( mostarda@fbk.eu )
32   * @version $Id: SerializationManager.java 2 2010-11-20 17:45:05Z michele.mostarda $
33   */
34  public class SerializationManager {
35  
36      class InternalSerializationContext implements SerializationContext {
37  
38          private Map<Object, Identifier> visitedObjects;
39  
40          InternalSerializationContext(Map<Object, Identifier> vo) {
41              visitedObjects = vo;
42          }
43  
44          public Map<Object, Identifier> getSerializedObjects() {
45              return visitedObjects;
46          }
47  
48          public Serializer getSerializerFor(Class clazz, Annotation[] annotations) throws SerializationException {
49              return SerializationManager.this.getSerializerFor(clazz, annotations);
50          }
51       
52          public Identifier serialize(SerializationContext context, Object object, Annotation[] annotations, TripleSet buffer)
53          throws SerializationException {
54              return SerializationManager.this.serialize(context, object, annotations, buffer);
55          }
56      }
57  
58      private static final Identifier NULL = new Identifier(BaseSerializer.NULL_VALUE, Identifier.Type.literal);
59  
60      private List<CriteriaSerializer> serializers;
61  
62      public SerializationManager() {
63          serializers = new ArrayList<CriteriaSerializer>();
64          configureDefaultSerializers();
65      }
66  
67      public boolean addSerializer(CriteriaSerializer serializer) {
68          if(serializer == null) {
69              throw new NullPointerException("serializer cannot be null.");
70          }
71          return serializers.add(serializer);
72      }
73  
74      public boolean removeSerializer(Serializer serializer) {
75          return serializers.remove(serializer);
76      }
77  
78      public Serializer getSerializerFor(Class clazz, Annotation[] annotations) throws SerializationException {
79          for(CriteriaSerializer serializer : serializers) {
80              if(serializer.acceptClass(clazz, annotations)) {
81                  return serializer;
82              }
83          }
84          throw new SerializationException(String.format("Cannot find serialized for %s", clazz.getName()) );
85      }
86  
87      public Identifier serialize(
88              SerializationContext context,
89              Object obj,
90              Annotation[] annotations,
91              TripleSet buffer
92      ) throws SerializationException {
93  
94          // Serialization of null returns a NULL representation.
95          if(obj == null) {
96              return NULL;
97          }
98  
99          final Map<Object, Identifier> visitedObjects = context.getSerializedObjects();
100 
101         // If the object to serialize has been already visited then the old result is returned.
102         Identifier identifier;
103         identifier = visitedObjects.get(obj);
104         if(identifier != null) {
105             return identifier;
106         }
107 
108         // The object is serialized and recorded.
109         Serializer serializer = getSerializerFor( obj.getClass(), annotations );
110 
111         if(serializer.isComplex()) {
112             identifier = serializer.getIdentifier(context, obj, annotations, buffer);
113             visitedObjects.put(obj, identifier);
114             serializer.serialize(context, obj, annotations, buffer);
115         } else {
116             identifier = serializer.serialize(context, obj, annotations, buffer);
117             visitedObjects.put(obj, identifier);
118         }
119 
120         return identifier;
121     }
122 
123     public void serializeObject(Object obj, Annotation[] annotations, TripleSet buffer) throws SerializationException {
124         Map<Object, Identifier> visitedObjects = new HashMap<Object, Identifier>();
125         serialize( new InternalSerializationContext(visitedObjects) , obj, annotations, buffer);
126     }
127 
128     public TripleBuffer serializeObject(Object obj, Annotation[] annotations) throws SerializationException {
129         TripleBuffer buffer = new TripleBuffer();
130         serializeObject(obj, annotations, buffer);
131         return buffer;
132     }
133 
134     public TripleBuffer serializeObject(Object obj) throws SerializationException {
135         TripleBuffer buffer = new TripleBuffer();
136         serializeObject(obj, BaseSerializer.getAnnotations(obj), buffer);
137         return buffer;
138     }
139 
140     /**
141      * Configures the default serializers in the right order.
142      */
143     private void configureDefaultSerializers() {
144 
145         // Static serializers.
146         addSerializer( new StaticCollectionSerializer() );
147         addSerializer( new StaticMapSerializer() );
148         addSerializer( new StaticBeanSerializer() );
149 
150         // Instance serializers.
151         addSerializer( new CollectionSerializer() );
152         addSerializer( new MapSerializer()        );
153         addSerializer( new EnumSerializer()       );
154         addSerializer( new BeanSerializer()       );
155 
156         // Primitive serializers.
157         addSerializer( new PrimitiveSerializer() );
158     }
