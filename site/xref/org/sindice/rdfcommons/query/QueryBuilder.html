17  package org.sindice.rdfcommons.query;
18  
19  import org.sindice.rdfcommons.storage.ResultSet;
20  import org.sindice.rdfcommons.storage.SparqlEndPoint;
21  import org.sindice.rdfcommons.storage.SparqlEndpointException;
22  import org.sindice.rdfcommons.model.TripleSet;
23  import org.sindice.rdfcommons.storage.TripleStorage;
24  
25  /**
26   * Defines a list of matches.
27   *
28   * @author Michele Mostarda ( mostarda@fbk.eu )
29   * @version $Id: QueryBuilder.java 24 2010-12-01 23:47:33Z michele.mostarda $
30   */
31  public class QueryBuilder {
32  
33      /**
34       * Internal chain of matches.
35       */
36      private MatchChain matchChain;
37  
38      /**
39       * Query context.
40       */
41      private QueryContext context;
42  
43      /**
44       * Constructor.
45       */
46      public QueryBuilder() {
47          context = new QueryContext();
48          matchChain = new MatchChain(context);
49      }
50  
51      /**
52       * @return the internal query context.
53       */
54      public QueryContext getContext() {
55          return context;
56      }
57  
58      /**
59       * Adds a query criteria.
60       *
61       * @param subCriteria
62       * @param predCriteria
63       * @param objCriteria
64       */
65      public QueryBuilder addQuery(String subCriteria, String predCriteria, String objCriteria) {
66          addQuery( new Match(context, subCriteria, predCriteria, objCriteria) );
67          return this;
68      }
69  
70      /**
71       * Removes a query criteria.
72       *
73       * @param subCriteria
74       * @param predCriteria
75       * @param objCriteria
76       */
77      public QueryBuilder removeQuery(String subCriteria, String predCriteria, String objCriteria) {
78          removeQuery( new Match(context, subCriteria, predCriteria, objCriteria) );
79          return this;
80      }
81  
82      /**
83       * Clears the chain query content.
84       */
85      public QueryBuilder clear() {
86          matchChain.clear();
87          return this;
88      }
89  
90      /**
91       * @return the <i>SPARQL</i> representation of the built query.
92       */
93      public String toSparqlSelect() {
94          return matchChain.toSparqlTriplePattern();
95      }
96  
97      /**
98       * Executes the built query on the given triple storage.
99       *
100      * @param ts the input storage.
101      * @return the obtained result set.
102      * @throws org.sindice.rdfcommons.storage.SparqlEndpointException if the triple storage doesn't support a <i>SPARQL</i> endpoint.
103      */
104     public ResultSet execOnStorage(TripleStorage ts) throws SparqlEndpointException {
105         SparqlEndPoint se = ts.getSparqlEndPoint();
106         return se.processSelectQuery( matchChain.toSparqlTriplePattern() );
107     }
108 
109     /**
110      * Executes the built query on the given triple set.
111      *
112      * @param ts input triple set.
113      * @return result set.
114      */
115     public ResultSet execOnTriples(TripleSet ts) {
116         TripleSetQueryExecutor queryExecutor = new TripleSetQueryExecutor();
117         return queryExecutor.filter(matchChain, ts);
118     }
119 
120     public String[] getVars() {
121         return matchChain.getVars();
122     }
123 
124     /**
125      * Adds a query criteria triple.
126      * 
127      * @param query
128      */
129     protected void addQuery(Match query) {
130         if(query == null) {
131             throw new IllegalArgumentException();
132         }
133         matchChain.add(query);
134     }
135 
136     /**
137      * Removes a query criteria triple.
138      *
139      * @param query
140      */
141     protected void removeQuery(Match query) {
142         if(query == null) {
143             throw new IllegalArgumentException();
144         }
145         matchChain.remove(query);
146     }
147 
148 }
