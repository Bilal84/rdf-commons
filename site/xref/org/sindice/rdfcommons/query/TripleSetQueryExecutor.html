41      /**
42       * Creates a match between a triple, the previous match and the current one.
43       * 
44       * @param t triple pattern.
45       * @param previous previous match
46       * @param current current match.
47       * @return new {@link org.sindice.rdfcommons.query.Match} instance.
48       */
49      protected static Match createMatch(Triple t, Match previous, Match current) {
50          if(previous == null) {
51              return current;
52          }
53          String[] currVars = current.getVars();
54          String value  = null;
55          String sub = null;
56          String pred = null;
57          String obj = null;
58          for(String currVar : currVars) {
59              int prevVarIndex = previous.getVarIndex(currVar);
60              int currVarIndex = current.getVarIndex(currVar);
61              switch (prevVarIndex) {
62                  case -1:
63                      value = currVar;
64                      break;
65                  case 0:
66                      value = t.getSubject();
67                      break;
68                  case 1:
69                      value = t.getPredicate();
70                      break;
71                  case 2:
72                      value = t.getObjectAsString();
73                      break;
74              }
75              switch (currVarIndex) {
76                  case 0:
77                      sub = value;
78                      break;
79                  case 1:
80                      pred = value;
81                      break;
82                  case 2:
83                      obj = value;
84                      break;
85              }
86          }
87          return new Match(current.getQueryContext(), sub, pred, obj);
88      }

90      /**
91       * Progresses an index array starting from the <i>begin</i> pointer, returning <code>-1</code>
92       * whether the progress sequence has been ended.
93       *
94       * @param begin pointer with the next index to be increased.
95       * @param indexes list of indexes.
96       * @param sets list of sets providing the upper bounds of every index.
97       * @return the next index to be updated.
98       */
99      protected static int progressIndex(int begin, int[] indexes, TripleSet[] sets) {
100         int i;
101         for(i = begin; i < indexes.length; i++) {
102             if(indexes[i] < sets[i].getSize() - 1) {
103                 indexes[i]++;
104                 return begin;
105             } else {
106                 indexes[i] = 0;
107             }
108         }
109         return i == indexes.length ? -1 : i;
110     }

113     /**
114      * Multiplies a list of sets generated by a list of matches and return the result inside the given result set.
115      *
116      * @param matches
117      * @param sets
118      * @param rs
119      */
120     protected static void multiply(Match[] matches, TripleSet[] sets, InMemoryResultSet rs) {
121         if(matches.length != sets.length) {
122             throw new IllegalArgumentException();
123         }
124 
125         int[] indexes = new int[sets.length];
126         int cursor = 0;
127         Match match;
128         List<InMemoryResultSet.VarEntry> entries = new ArrayList<InMemoryResultSet.VarEntry>();
129         while(cursor >= 0) {
130 
131             entries.clear();
132             Set<String> processedVars = new HashSet<String>();
133             for(int i = 0; i < matches.length; i++) {
134                 match = matches[i];
135                 List<Triple> triples = sets[i].getTriples();  // TODO: this MUST be optimized.
136                 Triple triple = triples.get(indexes[i]);
137 
138                 if(match.isSubVar()) {
139                     if( ! processedVars.contains(match.getSub()) ) {
140                         entries.add( new InMemoryResultSet.VarEntry(triple.getSubject()) );
141                         processedVars.add(match.getSub());
142                     }
143                 }
144                 if(match.isPredVar()) {
145                     if( ! processedVars.contains(match.getPred()) ) {
146                         entries.add( new InMemoryResultSet.VarEntry(triple.getPredicate()));
147                         processedVars.add(match.getPred());
148                     }
149                 }
150                 if(match.isObjVar()) {
151                     if( ! processedVars.contains(match.getObj()) ) {
152                         if(triple.isObjLiteral()) {
153                             entries.add(
154                                     new InMemoryResultSet.VarEntry(
155                                         triple.getObject(),
156                                         ResultSet.VariableType.LITERAL
157                                     )
158                             );
159                         } else {
160                             entries.add( new InMemoryResultSet.VarEntry(triple.getObjectAsString()) );
161                         }
162                         processedVars.add(match.getObj());
163                     }
164                 }
165 
166             }
167 
168             rs.addResult( entries.toArray( new InMemoryResultSet.VarEntry[entries.size()] ) );
169 
170             cursor = progressIndex(cursor, indexes, sets);
171         }
172     }

174     protected ResultSet filter(MatchChain qry, TripleSet in) {
175         if(qry.isEmpty()) {
176             throw new IllegalArgumentException("qry cannot be empty.");
177         }
178 
179         InMemoryResultSet rs = new InMemoryResultSet( qry.getVars() );
180 
181         if(in.getSize() == 0) {
182             return rs;
183         }
184 
185         // Creating initial results.
186         TripleSet matchSet = in.getTriples( new InternalFilter( qry.get(0) ) );
187         Stack<TripleSet> partialResults = new Stack<TripleSet>();
188         for(Triple triple : matchSet) {
189             partialResults.push( new TripleBuffer(triple) );
190         }
191 
192         Match previous = qry.get(0);
193         Match currentMatch;
194         TripleSet nextPartial;
195         while( ! partialResults.isEmpty() ) {
196 
197             nextPartial = partialResults.peek();
198 
199             for(Triple triple : nextPartial) {
200 
201                 boolean completed = true;
202                 for(int i = 1; i < qry.size(); i++) {
203 
204                     currentMatch = qry.get(i);
205 
206                     Match newMatch = createMatch(triple, previous, currentMatch);
207                     matchSet = in.getTriples( new InternalFilter(newMatch) );
208 
209                     if(matchSet.getSize() == 0) {
210                         completed = false;
211                         // Removing incomplete results.
212                         for(int j = 1; j < i; j++) {
213                             partialResults.pop();
214                         }
215                         break;
216                     }
217 
218                     partialResults.push(matchSet);
219 
220                 }
221 
222                 if(completed) {
223                     TripleSet[] sets = new TripleSet[qry.size()];
224                     for(int i = qry.size() - 1; i >= 0; i--) {
225                         sets[i] = partialResults.pop();
226                     }
227                     multiply(qry.getMatches(), sets, rs);
228                 }
229 
230             }
231 
232         }
233 
234         return rs;
235     }

237     class InternalFilter implements TripleFilter {
238 
239         private Match match;
240 
241         InternalFilter(Match m) {
242             match = m;
243         }
244 
245         public boolean acceptTriple(Triple triple) {
246             return
247             ( match.isSubVar()  || triple.getSubject().equals( match.getSub())   )
248                     &&
249             ( match.isPredVar() || triple.getPredicate().equals(match.getPred()) )
250                     &&
251             ( match.isObjVar()  || triple.getObject().equals( match.getObj())    );
252 
253         }
254     }
