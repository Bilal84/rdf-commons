247     private void containsStaticFakeBean(StaticFakeBean original, TripleBuffer tripleBuffer) {
248         assert tripleBuffer.containsTriple(
249                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean",
250                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean#field1",
251                 "field1 value", true
252         );
253         assert tripleBuffer.containsTriple(
254                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean",
255                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean#field2",
256                 true, true
257         );
258         assert tripleBuffer.containsTriple(
259                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean",
260                 "http://specific/property#field3",
261                 1, true
262         );
263         assert tripleBuffer.containsTriple(
264                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean",
265                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean#field4",
266                 original.getField4(), true
267         );
268         assert ! tripleBuffer.containsTriple(
269                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean",
270                 "http://org.sindice.rdfcommons.beanmapper/StaticFakeBean#field5",
271                 "this should be skipped.", true
272         ) : "Field5 is expected to be skipped.";
273 
274     }

276     private void containsFakeBean(FakeBean original, TripleBuffer tripleBuffer) throws MalformedURLException {
277         TripleSet ts = tripleBuffer.getTriples( new TripleFilter(){
278             public boolean acceptTriple(Triple triple) {
279                 return triple.getObject().equals("http://org.sindice.rdfcommons.beanmapper/FakeBean");
280             }
281         });
282         assert ts.getSize() == 1;
283         final String instanceURL = ts.getTriples().get(0).getSubject();
284         assert tripleBuffer.containsTriple(
285             instanceURL,
286             "http://org.sindice.rdfcommons.beanmapper/FakeBean#field1",
287             "value1", true
288         );
289         assert tripleBuffer.containsTriple(
290             instanceURL,
291             "http://org.sindice.rdfcommons.beanmapper/FakeBean#field2",
292             true, true
293         );
294         assert tripleBuffer.containsTriple(
295             instanceURL,
296             "http://org.sindice.rdfcommons.beanmapper/FakeBean#field3",
297             11, true
298         );
299         assert tripleBuffer.containsTriple(
300             instanceURL,
301             "http://org.sindice.rdfcommons.beanmapper/FakeBean#field4",
302             original.getField4(), true
303         );
304 
305         assert tripleBuffer.containsTriple(
306             instanceURL,
307             "http://org.sindice.rdfcommons.beanmapper/FakeBean#field5",
308             new URL("http://www.afakeulr.com/aqua"), true
309         );
310 
311         assert tripleBuffer.containsTriple(
312             instanceURL,
313             "http://org.sindice.rdfcommons.beanmapper/FakeBean#field6",
314             3.1415f, true
315         );
